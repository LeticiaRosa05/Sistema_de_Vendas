Consultas SQL

- Visualização de todas as vendas > automaticamente separadas por forma de pagamento
SELECT * FROM VENDAS
GROUP BY FP
ORDER BY EXTRACT(MONTH FROM DTVENDA); >>melhor mês ou dia?

- Visualização de acordo com forma de pagamento
SELECT FP FROM VENDAS
WHERE FP = 'Dinheiro';

SELECT FP FROM VENDAS
WHERE FP = 'Cartão'
GROUP BY DEB, CRED; >> talvez seja melhor criar uma tabela para cartão e por como FK

SELECT FP FROM VENDAS
WHERE FP = 'Pix';

- Visualização por dia com valor total vendido no dia
SELECT EXTRACT(DAY FROM DTVENDA) FROM VENDAS
WHERE (SELECT * FP FROM VENDAS
    GROUP BY FP
    SUM(valorTotal) AS 'Valor total vendido'
    WHERE FP = 'Cartão' >> tabela separada
    GROUP BY DEB, CRED);

- Visualização por mês(com valor total vendido por FP) com separação de dias e valor total de cada um
SELECT EXTRACT(MONTH FROM DTVENDA) FROM VENDAS
WHERE(
    SELECT * FP FROM VENDAS
    GROUP BY FP
    SUM(valorTotal) AS 'Valor total vendido'
    WHERE FP = 'Cartão'
    GROUP BY DEB, CRED;
SELECT 'Dias' || EXTRACT(DAY FROM DTVENDA) || ' ' || 'Valor total vendido' || SUM(valorTotal) FROM VENDAS);

>> Balanço do primeiro semestre
SELECT FP, SUM(valorTotal)
FROM VENDAS
WHERE EXTRACT(MONTH FROM DTVENDA) IN (1,2,3,4,5,6);
GROUP BY FP
WHERE FP = 'Cartão'
GROUP BY DEB, CRED;

>> Balanço do segundo semestre
SELECT FP, SUM(valorTotal)
FROM VENDAS
WHERE EXTRACT(MONTH FROM DTVENDA) IN (7,8,9,10,11,12);
GROUP BY FP
WHERE FP = 'Cartão'
GROUP BY DEB, CRED;
