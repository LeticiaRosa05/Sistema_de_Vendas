3- Visualização por dia com valor total vendido no dia
SELECT EXTRACT(DAY FROM DTVENDA) FROM VENDAS
WHERE (SELECT * FP FROM VENDAS
    GROUP BY FP
    SUM(valorTotal) AS 'Valor total vendido'
    WHERE FP = 'Cartão' >> tabela separada
    GROUP BY DEB, CRED);

4- Visualização por mês(com valor total vendido por FP) com separação de dias e valor total de cada um
SELECT EXTRACT(MONTH FROM DTVENDA), 'Dia' || EXTRACT(DAY FROM DTVENDA) || ' ' || 'Valor total vendido' || SUM(valorTotal) FROM VENDAS
WHERE(SELECT * FP FROM VENDAS
    GROUP BY FP
    SUM(valorTotal) AS 'Valor total vendido'
    WHERE FP = 'Cartão'
    GROUP BY DEB, CRED);

WHERE(SELECT * FP FROM VENDAS
    GROUP BY FP
    SUM(valorTotal) AS 'Valor total vendido'
    WHERE FP = 'Cartão'
    GROUP BY DEB, CRED);
